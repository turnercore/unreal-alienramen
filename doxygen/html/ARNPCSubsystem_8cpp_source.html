117static FARNpcRelationshipState* FindOrAddNpcState(UARNPCSubsystem* Subsystem, UARSaveGame* SaveGame, const FGameplayTag NpcTag)
118{
119 if (!SaveGame || !NpcTag.IsValid())
120 {
121 return nullptr;
122 }
123
124 if (FARNpcRelationshipState* Existing = FindNpcState(SaveGame, NpcTag))
125 {
126 return Existing;
127 }
128
129 FARNpcRelationshipState& Added = SaveGame->NpcRelationshipStates.AddDefaulted_GetRef();
130 Added.NpcTag = NpcTag;
131
132 if (UContentLookupSubsystem* Lookup = GetLookupSubsystem(Subsystem))
133 {
134 FARNpcDefinitionRow Def;
135 if (ResolveNpcDefinition(Lookup, NpcTag, Def))
136 {
137 Added.LoveRating = FMath::Max(0, Def.StartingLoveRating);
138 Added.CurrentWantTag = Def.InitialWantTag;
139 Added.bCurrentWantSatisfied = false;
140 }
141 }
142
143 return &Added;

146bool UARNPCSubsystem::SubmitNpcRamenDelivery(FGameplayTag NpcTag, FGameplayTag DeliveredRamenTag, bool& bOutAccepted)
147{
148 bOutAccepted = false;
149
150 if (!IsAuthorityWorld_NPC(GetWorld()) || !NpcTag.IsValid() || !DeliveredRamenTag.IsValid())
151 {
152 return false;
153 }
154
155 UARSaveSubsystem* SaveSubsystem = GetSaveSubsystem(this);
156 UARSaveGame* SaveGame = SaveSubsystem ? SaveSubsystem->GetCurrentSaveGame() : nullptr;
157 if (!SaveGame)
158 {
159 return false;
160 }
161
162 FARNpcRelationshipState* State = FindOrAddNpcState(this, SaveGame, NpcTag);
163 if (!State)
164 {
165 return false;
166 }
167
168 if (!State->CurrentWantTag.IsValid() || !State->CurrentWantTag.MatchesTagExact(DeliveredRamenTag))
169 {
170 return true;
171 }
172
173 FARNpcDefinitionRow Def;
174 if (UContentLookupSubsystem* Lookup = GetLookupSubsystem(this))
175 {
176 ResolveNpcDefinition(Lookup, NpcTag, Def);
177 }
178
179 State->LoveRating = FMath::Max(0, State->LoveRating + FMath::Max(1, Def.LoveIncreaseOnWantDelivery));
180 State->bCurrentWantSatisfied = true;
181 bOutAccepted = true;
182 SaveSubsystem->MarkSaveDirty();
183
184 RefreshNpcTalkableState(NpcTag);
185 return true;
186}

230bool UARNPCSubsystem::RefreshNpcTalkableState(FGameplayTag NpcTag)
