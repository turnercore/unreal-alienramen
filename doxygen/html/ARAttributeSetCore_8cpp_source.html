53void UARAttributeSetCore::PreAttributeChange(const FGameplayAttribute& Attribute, float& NewValue)
54{
55 // Clamp common "max" attributes to non-negative
56 if (Attribute == GetMaxHealthAttribute() ||
57 Attribute == GetMaxShieldAttribute() ||
58 Attribute == GetMaxJetpackFuelAttribute() ||
59 Attribute == GetMaxSpiceAttribute() ||
60 Attribute == GetMaxHatEnergyAttribute() ||
61 Attribute == GetMaxAmmoAttribute() ||
62 Attribute == GetSecondaryMaxAmmoAttribute() ||
63 Attribute == GetSpecialMaxAmmoAttribute())
64 {
65 NewValue = ClampNonNegative(NewValue);
66 }
67
68 // Clamp [0..1] style attributes
69 if (Attribute == GetCritChanceAttribute())
70 {
71 NewValue = Clamp01(NewValue);
72 }
73
74 if (Attribute == GetDropChanceAttribute())
75 {
76 NewValue = Clamp01(NewValue);
77 }
78
79 // Multipliers that should never go negative
80 if (Attribute == GetDamageTakenMultiplierAttribute() ||
81 Attribute == GetHealingReceivedMultiplierAttribute() ||
82 Attribute == GetHealingDealtMultiplierAttribute() ||
83 Attribute == GetSpiceGainMultiplierAttribute() ||
84 Attribute == GetSpreadMultiplierAttribute() ||
85 Attribute == GetMeatDropMultiplierAttribute() ||
86 Attribute == GetScrapDropMultiplierAttribute())
87 {
88 NewValue = FMath::Max(0.0f, NewValue);
89 }
90
91 if (Attribute == GetDropAmountAttribute())
92 {
93 NewValue = FMath::Max(0.0f, NewValue);
94 }
95
96 // Keep current values within new max values
97 if (Attribute == GetMaxHealthAttribute())
98 {
99 SetHealth(FMath::Clamp(GetHealth(), 0.0f, NewValue));
100 }
101 else if (Attribute == GetMaxShieldAttribute())
102 {
103 SetShield(FMath::Clamp(GetShield(), 0.0f, NewValue));
104 }
105 else if (Attribute == GetMaxJetpackFuelAttribute())
106 {
107 SetJetpackFuel(FMath::Clamp(GetJetpackFuel(), 0.0f, NewValue));
108 }
109 else if (Attribute == GetMaxSpiceAttribute())
110 {
111 SetSpice(FMath::Clamp(GetSpice(), 0.0f, NewValue));
112 }
113 else if (Attribute == GetMaxHatEnergyAttribute())
114 {
115 SetHatEnergy(FMath::Clamp(GetHatEnergy(), 0.0f, NewValue));
116 }
117 else if (Attribute == GetMaxAmmoAttribute())
118 {
119 SetAmmo(FMath::Clamp(GetAmmo(), 0.0f, NewValue));
120 }
121 else if (Attribute == GetSecondaryMaxAmmoAttribute())
122 {
123 SetSecondaryAmmo(FMath::Clamp(GetSecondaryAmmo(), 0.0f, NewValue));
124 }
125 else if (Attribute == GetSpecialMaxAmmoAttribute())
126 {
127 SetSpecialAmmo(FMath::Clamp(GetSpecialAmmo(), 0.0f, NewValue));
128 }

131void UARAttributeSetCore::PostGameplayEffectExecute(const FGameplayEffectModCallbackData& Data)
132{
133 const FGameplayAttribute& Attr = Data.EvaluatedData.Attribute;
134
135 // IncomingDamage is a transient meta-attribute consumed into shield/health.
136 if (Attr == GetIncomingDamageAttribute())
137 {
138 const float PendingDamage = FMath::Max(0.f, GetIncomingDamage());
139 SetIncomingDamage(0.f);
140
141 if (PendingDamage > 0.f)
142 {
143 float RemainingDamage = PendingDamage * FMath::Max(0.f, GetDamageTakenMultiplier());
144
145 if (GetMaxShield() > 0.f && GetShield() > 0.f)
146 {
147 const float NewShield = FMath::Max(0.f, GetShield() - RemainingDamage);
148 RemainingDamage = FMath::Max(0.f, RemainingDamage - GetShield());
149 SetShield(NewShield);
150 }
151
152 if (RemainingDamage > 0.f)
153 {
154 SetHealth(FMath::Clamp(GetHealth() - RemainingDamage, 0.f, GetMaxHealth()));
155 }
156 }
157
158 return;
159 }
160
161 // Clamp currents to [0..Max]
162 if (Attr == GetHealthAttribute())
163 {
164 SetHealth(FMath::Clamp(GetHealth(), 0.0f, GetMaxHealth()));
165 }
166 else if (Attr == GetShieldAttribute())
167 {
168 SetShield(FMath::Clamp(GetShield(), 0.0f, GetMaxShield()));
169 }
170 else if (Attr == GetJetpackFuelAttribute())
171 {
172 SetJetpackFuel(FMath::Clamp(GetJetpackFuel(), 0.0f, GetMaxJetpackFuel()));
173 }
174 else if (Attr == GetSpiceAttribute())
175 {
176 SetSpice(FMath::Clamp(GetSpice(), 0.0f, GetMaxSpice()));
177 }
178 else if (Attr == GetHatEnergyAttribute())
179 {
180 SetHatEnergy(FMath::Clamp(GetHatEnergy(), 0.0f, GetMaxHatEnergy()));
181 }
182 else if (Attr == GetAmmoAttribute())
183 {
184 SetAmmo(FMath::Clamp(GetAmmo(), 0.0f, GetMaxAmmo()));
185 }
186 else if (Attr == GetSecondaryAmmoAttribute())
187 {
188 SetSecondaryAmmo(FMath::Clamp(GetSecondaryAmmo(), 0.0f, GetSecondaryMaxAmmo()));
189 }
190 else if (Attr == GetSpecialAmmoAttribute())
191 {
192 SetSpecialAmmo(FMath::Clamp(GetSpecialAmmo(), 0.0f, GetSpecialMaxAmmo()));
193 }
194
195 // Clamp [0..1] attributes post-mod too
196 if (Attr == GetCritChanceAttribute())
197 {
198 SetCritChance(Clamp01(GetCritChance()));
199 }
200
201 if (Attr == GetDropChanceAttribute())
202 {
203 SetDropChance(Clamp01(GetDropChance()));
204 }
205
206 // Non-negative clamps
207 if (Attr == GetDamageTakenMultiplierAttribute())
208 {
209 SetDamageTakenMultiplier(FMath::Max(0.0f, GetDamageTakenMultiplier()));
210 }
211 else if (Attr == GetHealingReceivedMultiplierAttribute())
212 {
213 SetHealingReceivedMultiplier(FMath::Max(0.0f, GetHealingReceivedMultiplier()));
214 }
215 else if (Attr == GetHealingDealtMultiplierAttribute())
216 {
217 SetHealingDealtMultiplier(FMath::Max(0.0f, GetHealingDealtMultiplier()));
218 }
219 else if (Attr == GetSpiceGainMultiplierAttribute())
220 {
221 SetSpiceGainMultiplier(FMath::Max(0.0f, GetSpiceGainMultiplier()));
222 }
223 else if (Attr == GetSpreadMultiplierAttribute())
224 {
225 SetSpreadMultiplier(FMath::Max(0.0f, GetSpreadMultiplier()));
226 }
227 else if (Attr == GetMeatDropMultiplierAttribute())
228 {
229 SetMeatDropMultiplier(FMath::Max(0.0f, GetMeatDropMultiplier()));
230 }
231 else if (Attr == GetScrapDropMultiplierAttribute())
232 {
233 SetScrapDropMultiplier(FMath::Max(0.0f, GetScrapDropMultiplier()));
234 }
235
236 if (Attr == GetDropAmountAttribute())
237 {
238 SetDropAmount(FMath::Max(0.0f, GetDropAmount()));
239 }
240
241 // Optional: keep crit multiplier sensible (>= 1)
242 if (Attr == GetCritMultiplierAttribute())
243 {
244 SetCritMultiplier(FMath::Max(1.0f, GetCritMultiplier()));
245 }
246}
